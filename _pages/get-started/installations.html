Goldilocks-CSS works without a package manager or dependencies! For this beta phase, there are currently 2 ways to get started with Goldilocks-CSS: download from Github or use a CDN.

Install manually

Download Goldilocks-CSS files from Github (use link in navbar) and link one of the following files into the head of your website:

Core includes Layouts, Common Elements and Helpers

<link rel="stylesheet" href="./dist/goldilocks.core.min.css">

Core Plus Popular Add-on Elements and Components

<link rel="stylesheet" href="./dist/goldilocks.core-extended.min.css">

Core and Extended Items Plus Secondary Add-on Elements and Components

<link rel="stylesheet" href="./dist/goldilocks.all.min.css">

Install from CDN

Alternatively, you can use jsdelivr CDN to link to Goldilocks-CSS. Use only one link for the package that you need.

<link rel="stylesheet" href="https://cdn.jsdelivr.net/gh/zaphodbb-pm/goldilocks-css/dist/goldilocks.core.min.css"> <link rel="stylesheet" href="https://cdn.jsdelivr.net/gh/zaphodbb-pm/goldilocks-css/dist/goldilocks.core-extended.min.css"> <link rel="stylesheet" href="https://cdn.jsdelivr.net/gh/zaphodbb-pm/goldilocks-css/dist/goldilocks.all.min.css">

Starter HTML template

An example starter template that sets up the basic page structure. This typical for Github Pages use with the Jekyll template language. Included is some Javascript to load the icon svg files. 


            <!doctype html>
            <html class="no-js" lang="en">
                <head>
                    <meta charset="utf-8">
                    <title>{{page.title}} for Goldilocks CSS</title>
                    <meta name="viewport" content="width=device-width, initial-scale=1">
                    <meta name="description" content="Goldilocks-CSS is a lightweight CSS framework based on grid that is HTML5 and CSS3 standards compliant.
                    It is 100% responsive, fully modular, no javascript required and available for free on an MIT license.">
                    <meta name="robots" content="index, follow">

                    <link rel="icon" href="{{site.url}}/favicon.png">
                    <link rel="stylesheet" href="https://cdn.jsdelivr.net/gh/zaphodbb-pm/goldilocks-css/dist/goldilocks.all.min.css" media="screen">
                </head>

                <body>
                    <div id="page-top" class="page-wrapper">

                        <div id="page-main-nav" class="page-main-nav">
                            <nav class="navbar is-primary has-sidebar" aria-label="Main Site Navigation Bar">
                                <div class="navbar-left">
                                    <!--add your own logo -->
                                    <a href="{{site.url}}/home.html">
                                        <img src="{{site.url}}/img/logo.png" height="192" width="192" alt="Project Logo">
                                    </a>
                                </div>

                                <div class="navbar-right">
                                    <!-- add your links here -->
                                    <a class="icon-nav" href="https://github.com/zaphodbb-pm/goldilocks-css" rel="nofollow" aria-label="link to github">
                                        <svg class="icon"><use href="#icon-github"></use></svg>
                                    </a>
                                </div>
                            </nav>
                        </div>

                        <!-- include heading and main content -->
                        <header class="page-header">
                            <h1>{{page.title}}</h1>
                        </header>

                        <main class="main-content">
                            Content
                        </main>

                        <footer class="page-footer"></footer>
                    </div>

                    <!--
                        Loads svg based icons that can be referenced on any page.
                        Make sure that the icons definition file is publicly accessable.
                     -->

                    <script>
                        fetch("{{site.url}}/dist/icon-definitions.html")
                                .then( response => response.text() )
                                .then((data) => {
                                    let svgElements = document.createElement('div');
                                    svgElements.innerHTML = data;
                                    document.body.appendChild(svgElements);
                                })

                        /* OR use the alternative icon set */

                        fetch("{{site.url}}/dist/icon-la-definitions.html")
                                .then( response => response.text() )
                                .then((data) => {
                                    let svgElements = document.createElement('div');
                                    svgElements.innerHTML = data;
                                    document.body.appendChild(svgElements);
                                })
                    </script>

                </body>
            </html>
